\begin{exercise}[BH.5.28]
\begin{solution}
    $$P(|Y-\mu|<1.96 \sigma)=P(-1.96 \sigma<Y-\mu<1.96 \sigma)=P(Y-1.96 \sigma<\mu<Y+1.96 \sigma)
    $$
    shows that the random interval $(Y-1.96 \sigma, Y+1.96 \sigma)$ is as desired.
\end{solution}
\end{exercise}
