\begin{solution}
    \begin{enumerate}
	    \item By definition of expectation of discrete r.v.'s, $\mathbb{E}(X_1)=\sum_{i=1}^{6} i/6 =3.5$. 
		\item By linearity, $\mathbb{E}(\sum_{i=1}^4 X_i) =4\mathbb{E}(X_1)= 14$.  
	\end{enumerate}
\end{solution}
